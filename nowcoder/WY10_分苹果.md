# myCode
- 遍历：44ms	4744KB
```python
import sys

lines = sys.stdin.readlines()

n = lines[0].strip()
arr = lines[1].strip().split()
narr = list(map(int, arr))

sums = sum(narr)

def apple(narr,sums):
    num = 0
    if sums % len(narr) == 0:
        avg = sums / len(narr)
        for i in narr:
            if abs(avg-i) % 2 != 0:
                return -1
            elif i > avg:
                num += (i-avg) / 2
        return int(num)
    else:
        return -1

num = apple(narr,sums)
print(num)
```
